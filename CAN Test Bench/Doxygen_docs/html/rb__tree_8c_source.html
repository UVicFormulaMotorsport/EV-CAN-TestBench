90 rbnode *rbSuccessor(rbtree *rbt, rbnode *node)
91 {
92  rbnode *p;
93 
94  p = node->right;
95 
96  if (p != RB_NIL(rbt)) {
97  /* move down until we find it */
98  for ( ; p->left != RB_NIL(rbt); p = p->left) ;
99  } else {
100  /* move up until we find it or hit the root */
101  for (p = node->parent; node == p->right; node = p, p = p->parent) ;
102 
103  if (p == RB_ROOT(rbt))
104  p = NULL; /* not found */
105  }
106

110 /*
111  * apply func
112  * return non-zero if error
113  */
114 int rb_apply(rbtree *rbt, rbnode *node, int (*func)(void *, void *), void *cookie, enum rbtraversal order)
115 {
116  int err;
117 
118  if (node != RB_NIL(rbt)) {
119  if (order == PREORDER && (err = func(node->data, cookie)) != 0) /* preorder */
120  return err;
121  if ((err = rb_apply(rbt, node->left, func, cookie, order)) != 0) /* left */
122  return err;
123  if (order == INORDER && (err = func(node->data, cookie)) != 0) /* inorder */
124  return err;
125  if ((err = rb_apply(rbt, node->right, func, cookie, order)) != 0) /* right */
126  return err;
127  if (order == POSTORDER && (err = func(node->data, cookie)) != 0) /* postorder */
128  return err;
129  }
130 
131  return 0;
132 }

138 {

140

142 
143  /* tree x */
144  x->right = y->left;
145  if (x->right != RB_NIL(rbt))
146  x->right->parent = x;
147 
148  /* tree y */
149  y->parent = x->parent;
150  if (x == x->parent->left)
151  x->parent->left = y;
152  else
153  x->parent->right = y;
154 
155  /* assemble tree x and tree y */
156  y->left = x;
157  x->parent = y;

187 /*
188  * insert (or update) data
189  * return NULL if out of memory
190  */
191 rbnode *rbInsert(rbtree *rbt, void *data)
192 {
193  rbnode *current, *parent;
194  rbnode *new_node;
195 
196  /* do a binary search to find where it should be */
197 
198  current = RB_FIRST(rbt);
199  parent = RB_ROOT(rbt);
200 
201  while (current != RB_NIL(rbt)) {
202  int cmp;
203  cmp = rbt->compare(data, current->data);
204 
205  #ifndef RB_DUP
206  if (cmp == 0) {
207  rbt->destroy(current->data);
208  current->data = data;
209  return current; /* updated */
210  }
211  #endif
212 
213  parent = current;
214  current = cmp < 0 ? current->left : current->right;
215  }
216 
217  /* replace the termination NIL pointer with the new node pointer */
218 
219  current = new_node = (rbnode *) uvMalloc(sizeof(rbnode));
220  if (current == NULL)
221  return NULL; /* out of memory */
222 
223  current->left = current->right = RB_NIL(rbt);
224  current->parent = parent;
225  current->color = RED;
226  current->data = data;
227 
228  if (parent == RB_ROOT(rbt) || rbt->compare(data, parent->data) < 0)
229  parent->left = current;
230  else
231  parent->right = current;
232 
233  #ifdef RB_MIN
234  if (rbt->min == NULL || rbt->compare(current->data, rbt->min->data) < 0)
235  rbt->min = current;
236  #endif
237 
238  /*
239  * insertion into a red-black tree:
240  * 0-children root cluster (parent node is BLACK) becomes 2-children root cluster (new root node)
241  * paint root node BLACK, and done
242  * 2-children cluster (parent node is BLACK) becomes 3-children cluster
243  * done
244  * 3-children cluster (parent node is BLACK) becomes 4-children cluster
245  * done
246  * 3-children cluster (parent node is RED) becomes 4-children cluster
247  * rotate, and done
248  * 4-children cluster (parent node is RED) splits into 2-children cluster and 3-children cluster
249  * split, and insert grandparent node into parent cluster
250  */
251  if (current->parent->color == RED) {
252  /* insertion into 3-children cluster (parent node is RED) */
253  /* insertion into 4-children cluster (parent node is RED) */
254  insertRepair(rbt, current);
255  } else {
256  /* insertion into 0-children root cluster (parent node is BLACK) */
257  /* insertion into 2-children cluster (parent node is BLACK) */
258  /* insertion into 3-children cluster (parent node is BLACK) */
259  }
260 
261  /*
262  * the root is always BLACK
263  * insertion into 0-children root cluster or insertion into 4-children root cluster require this recoloring
264  */
265  RB_FIRST(rbt)->color = BLACK;
266 
267  rbt->count++;
268 
269  return new_node;
270 }

272 /*
273  * rebalance after insertion
274  * RB_ROOT(rbt) is always BLACK, thus never reach beyond RB_FIRST(rbt)
275  * after insertRepair, RB_FIRST(rbt) might be RED
276  */
277 void insertRepair(rbtree *rbt, rbnode *current)
278 {
279  rbnode *uncle;
280 
281  do {
282  /* current node is RED and parent node is RED */
283 
284  if (current->parent == current->parent->parent->left) {
285  uncle = current->parent->parent->right;
286  if (uncle->color == RED) {
287  /* insertion into 4-children cluster */
288 
289  /* split */
290  current->parent->color = BLACK;
291  uncle->color = BLACK;
292 
293  /* send grandparent node up the tree */
294  current = current->parent->parent; /* goto loop or break */
295  current->color = RED;
296  } else {
297  /* insertion into 3-children cluster */
298 
299  /* equivalent BST */
300  if (current == current->parent->right) {
301  current = current->parent;
302  rotateLeft(rbt, current);
303  }
304 
305  /* 3-children cluster has two representations */
306  current->parent->color = BLACK; /* thus goto break */
307  current->parent->parent->color = RED;
308  rotateRight(rbt, current->parent->parent);
309  }
310  } else {
311  uncle = current->parent->parent->left;
312  if (uncle->color == RED) {
313  /* insertion into 4-children cluster */
314 
315  /* split */
316  current->parent->color = BLACK;
317  uncle->color = BLACK;
318 
319  /* send grandparent node up the tree */
320  current = current->parent->parent; /* goto loop or break */
321  current->color = RED;
322  } else {
323  /* insertion into 3-children cluster */
324 
325  /* equivalent BST */
326  if (current == current->parent->left) {
327  current = current->parent;
328  rotateRight(rbt, current);
329  }
330 
331  /* 3-children cluster has two representations */
332  current->parent->color = BLACK; /* thus goto break */
333  current->parent->parent->color = RED;
334  rotateLeft(rbt, current->parent->parent);
335  }
336  }
337  } while (current->parent->color == RED);
338 }

340 /*
341  * delete node
342  * return NULL if keep is zero (already freed)
343  */
344 void *rbDelete(rbtree *rbt, rbnode *node, int keep)
345 {
346  rbnode *target, *child;
347  void *data;
348 
349  data = node->data;
350 
351  /* choose node's in-order successor if it has two children */
352 
353  if (node->left == RB_NIL(rbt) || node->right == RB_NIL(rbt)) {
354  target = node;
355 
356  #ifdef RB_MIN
357  if (rbt->min == target)
358  rbt->min = rbSuccessor(rbt, target); /* deleted, thus min = successor */
359  #endif
360  } else {
361  target = rbSuccessor(rbt, node); /* node->right must not be NIL, thus move down */
362 
363  node->data = target->data; /* data swapped */
364 
365  #ifdef RB_MIN
366  /* if min == node, then min = successor = node (swapped), thus idle */
367  /* if min == target, then min = successor, which is not the minimal, thus impossible */
368  #endif
369  }
370 
371  child = (target->left == RB_NIL(rbt)) ? target->right : target->left; /* child may be NIL */
372 
373  /*
374  * deletion from red-black tree
375  * 4-children cluster (RED target node) becomes 3-children cluster
376  * done
377  * 3-children cluster (RED target node) becomes 2-children cluster
378  * done
379  * 3-children cluster (BLACK target node, RED child node) becomes 2-children cluster
380  * paint child node BLACK, and done
381  *
382  * 2-children root cluster (BLACK target node, BLACK child node) becomes 0-children root cluster
383  * done
384  *
385  * 2-children cluster (BLACK target node, 4-children sibling cluster) becomes 3-children cluster
386  * transfer, and done
387  * 2-children cluster (BLACK target node, 3-children sibling cluster) becomes 2-children cluster
388  * transfer, and done
389  *
390  * 2-children cluster (BLACK target node, 2-children sibling cluster, 3/4-children parent cluster) becomes 3-children cluster
391  * fuse, paint parent node BLACK, and done
392  * 2-children cluster (BLACK target node, 2-children sibling cluster, 2-children parent cluster) becomes 3-children cluster
393  * fuse, and delete parent node from parent cluster
394  */
395  if (target->color == BLACK) {
396  if (child->color == RED) {
397  /* deletion from 3-children cluster (BLACK target node, RED child node) */
398  child->color = BLACK;
399  } else if (target == RB_FIRST(rbt)) {
400  /* deletion from 2-children root cluster (BLACK target node, BLACK child node) */
401  } else {
402  /* deletion from 2-children cluster (BLACK target node, ...) */
403  deleteRepair(rbt, target);
404  }
405  } else {
406  /* deletion from 4-children cluster (RED target node) */
407  /* deletion from 3-children cluster (RED target node) */
408  }
409 
410  if (child != RB_NIL(rbt))
411  child->parent = target->parent;
412 
413  if (target == target->parent->left)
414  target->parent->left = child;
415  else
416  target->parent->right = child;
417 
418  uvFree(target);
419 
420  /* keep or discard data */
421  if (keep == 0) {
422  rbt->destroy(data);
423  data = NULL;
424  }
425 
426  rbt->count --;
427 
428  return data;
429 }

431 /*
432  * rebalance after deletion
433  */
434 void deleteRepair(rbtree *rbt, rbnode *current)
435 {
436  rbnode *sibling;
437  do {
438  if (current == current->parent->left) {
439  sibling = current->parent->right;
440 
441  if (sibling->color == RED) {
442  /* perform an adjustment (3-children parent cluster has two representations) */
443  sibling->color = BLACK;
444  current->parent->color = RED;
445  rotateLeft(rbt, current->parent);
446  sibling = current->parent->right;
447  }
448 
449  /* sibling node must be BLACK now */
450 
451  if (sibling->right->color == BLACK && sibling->left->color == BLACK) {
452  /* 2-children sibling cluster, fuse by recoloring */
453  sibling->color = RED;
454  if (current->parent->color == RED) { /* 3/4-children parent cluster */
455  current->parent->color = BLACK;
456  break; /* goto break */
457  } else { /* 2-children parent cluster */
458  current = current->parent; /* goto loop */
459  }
460  } else {
461  /* 3/4-children sibling cluster */
462 
463  /* perform an adjustment (3-children sibling cluster has two representations) */
464  if (sibling->right->color == BLACK) {
465  sibling->left->color = BLACK;
466  sibling->color = RED;
467  rotateRight(rbt, sibling);
468  sibling = current->parent->right;
469  }
470 
471  /* transfer by rotation and recoloring */
472  sibling->color = current->parent->color;
473  current->parent->color = BLACK;
474  sibling->right->color = BLACK;
475  rotateLeft(rbt, current->parent);
476  break; /* goto break */
477  }
478  } else {
479  sibling = current->parent->left;
480 
481  if (sibling->color == RED) {
482  /* perform an adjustment (3-children parent cluster has two representations) */
483  sibling->color = BLACK;
484  current->parent->color = RED;
485  rotateRight(rbt, current->parent);
486  sibling = current->parent->left;
487  }
488 
489  /* sibling node must be BLACK now */
490 
491  if (sibling->right->color == BLACK && sibling->left->color == BLACK) {
492  /* 2-children sibling cluster, fuse by recoloring */
493  sibling->color = RED;
494  if (current->parent->color == RED) { /* 3/4-children parent cluster */
495  current->parent->color = BLACK;
496  break; /* goto break */
497  } else { /* 2-children parent cluster */
498  current = current->parent; /* goto loop */
499  }
500  } else {
501  /* 3/4-children sibling cluster */
502 
503  /* perform an adjustment (3-children sibling cluster has two representations) */
504  if (sibling->left->color == BLACK) {
505  sibling->right->color = BLACK;
506  sibling->color = RED;
507  rotateLeft(rbt, sibling);
508  sibling = current->parent->left;
509  }
510 
511  /* transfer by rotation and recoloring */
512  sibling->color = current->parent->color;
513  current->parent->color = BLACK;
514  sibling->left->color = BLACK;
515  rotateRight(rbt, current->parent);
516  break; /* goto break */
517  }
518  }
519  } while (current != RB_FIRST(rbt));
520 }
