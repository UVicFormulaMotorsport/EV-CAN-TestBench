62 static uint16_t MotorControllerSpinTest(void) {
63  uint8_t spin_command[2] = {0x00, 0x10}; // Command to spin at low RPM
64 
65  // Use CAN_TxHeaderTypeDef for message header
66  CAN_TxHeaderTypeDef txHeader;
67  txHeader.StdId = mc_default_settings.can_id_tx; // Set CAN ID
68  txHeader.IDE = CAN_ID_STD; // Use standard ID
69  txHeader.RTR = CAN_RTR_DATA; // Data frame
70  txHeader.DLC = 3; // Data length code
71 
72  // Transmit the message
73  if (HAL_CAN_AddTxMessage(&hcan2, &txHeader, (uint8_t[]){N_set, spin_command[0], spin_command[1]}, &TxMailbox) != HAL_OK) {
74  return 1; // Transmission error
75  }
76 
77  vTaskDelay(pdMS_TO_TICKS(500)); // Allow time for the motor to spin
78 
79  // Request actual motor speed
80  MC_Request_Data(N_actual);
81  if (!WaitFor_CAN_Response()) {
82  return 2; // Timeout occurred
83  }
84 
85  uint16_t actual_speed = (RxData[0] << 8) | RxData[1];
86  if (actual_speed < 0x10) { // If the motor did not reach the expected speed
87  return 3; // Motor failed to spin
88  }
89 
90  // Stop the motor
91  uint8_t stop_command[2] = {0x00, 0x00};
92  if (HAL_CAN_AddTxMessage(&hcan2, &txHeader, (uint8_t[]){N_set, stop_command[0], stop_command[1]}, &TxMailbox) != HAL_OK) {
93  return 4; // Transmission error
94  }
95 
96  return 0; // Success
97 }
